AgriLabour Point is a hyperlocal marketplace matching local farmers with skilled laborers and machinery rentals. Connect directly, schedule on interactive calendars, and perform secure mock transactions without any middleman commissions.

For Farmers

Secure support for your field and manage crop cutting timelines with local resources:

  • Hire Local Helpers Browse local worker profiles, list crop harvesting requirements, and inspect verified ratings.
  • Rent Tractors & Implements Rent tractors, tillers, or rotavators hyperlocally from nearby owners with verified operators.
  • Interactive Scheduling View laborer availability on a calendar, select preferred dates, and block dates to request booking.
  • Safe Mock Payouts Pay wages directly via simulated UPI gateways with built-in transaction logging history.

For Laborers & Machine Owners

Register your profile and machinery to secure local work requests and maximize income:

  • Publish Skilled Services Choose your crop skills (seed sowing, pesticide spraying, harvesting, driving) and highlight your profile.
  • Set Custom Rates Set direct pricing units per hour or per day. You receive 100% of your earnings with zero cuts.
  • Calendar Bookings Tracker View upcoming work reservations, manage dates, and update availability status automatically.
  • Build Customer Reputation Deliver quality work to gather stars and writeups, ensuring priority booking from top local landowners.

-- AgriLabour Point: Database Schema
-- Optimized for lightweight rural connectivity, hyper-local GIS indexing, and relational skill ratings.

CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "postgis"; -- Location queries

CREATE TYPE user_role AS ENUM ('farmer', 'laborer', 'machinery_owner');
CREATE TYPE listing_type AS ENUM ('labor', 'machinery');
CREATE TYPE booking_status AS ENUM ('pending', 'confirmed', 'cancelled', 'completed');

-- Users & Language Preferences
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    phone_number VARCHAR(15) UNIQUE NOT NULL,
    name VARCHAR(100),
    role user_role NOT NULL,
    preferred_language VARCHAR(5) DEFAULT 'en',
    geom GEOMETRY(Point, 4326), -- Geographic GPS point
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_users_geom ON users USING gist(geom);

-- Skill taxonomy lookup
CREATE TABLE skills (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tag VARCHAR(50) UNIQUE NOT NULL, -- e.g., 'tractor_driving', 'harvesting', 'spraying'
    name_en VARCHAR(100) NOT NULL,
    name_hi VARCHAR(100) NOT NULL,
    name_mr VARCHAR(100) NOT NULL
);

-- Many-to-Many Skill Match table
CREATE TABLE laborer_skills (
    laborer_id UUID REFERENCES users(id) ON DELETE CASCADE,
    skill_id UUID REFERENCES skills(id) ON DELETE CASCADE,
    experience_months INTEGER DEFAULT 0,
    PRIMARY KEY (laborer_id, skill_id)
);

-- Listings with transparent rates
CREATE TABLE listings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    owner_id UUID REFERENCES users(id) ON DELETE CASCADE,
    type listing_type NOT NULL,
    title VARCHAR(150) NOT NULL,
    description TEXT,
    price_amount DECIMAL(10, 2) NOT NULL,
    price_currency VARCHAR(3) DEFAULT 'INR',
    price_unit VARCHAR(10) DEFAULT 'day', -- 'day', 'hour'
    machinery_type VARCHAR(50),
    availability_status VARCHAR(20) DEFAULT 'available',
    geom GEOMETRY(Point, 4326),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_listings_geom ON listings USING gist(geom);

-- Calendar Bookings tracking availability
CREATE TABLE bookings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    listing_id UUID REFERENCES listings(id) ON DELETE CASCADE,
    farmer_id UUID REFERENCES users(id) ON DELETE CASCADE,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    status booking_status DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Relational skill-specific ratings
CREATE TABLE reviews (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    listing_id UUID REFERENCES listings(id) ON DELETE SET NULL,
    author_id UUID REFERENCES users(id) ON DELETE CASCADE,
    subject_id UUID REFERENCES users(id) ON DELETE CASCADE,
    skill_id UUID REFERENCES skills(id) ON DELETE SET NULL,
    rating INTEGER CHECK (rating >= 1 AND rating <= 5) NOT NULL,
    comment TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_reviews_subject ON reviews(subject_id);

-- Logic Flow for Multilingual Search Index
-- Translates query to universal identifier (tag) then executes database query:

WITH resolved_tags AS (
    SELECT 'tractor_driving' AS tag -- Resolved from input "ट्रैक्टर", "कापणी", "tractor" via mapper
),
matched_listings AS (
    SELECT l.*, 
           -- PostGIS distance computation in meters
           ST_Distance(l.geom, ST_MakePoint(73.8567, 18.5204)::geography) AS distance_meters
    FROM listings l
    -- ManyToMany skill relation join or simple direct tag matching
    WHERE l.machinery_type = 'tractor' 
       OR l.id IN (
           SELECT listing_id FROM reviews r 
           INNER JOIN skills s ON r.skill_id = s.id 
           WHERE s.tag = (SELECT tag FROM resolved_tags)
       )
)
SELECT * FROM matched_listings 
WHERE distance_meters <= 10000 -- Hyperlocal boundary (10km)
ORDER BY distance_meters ASC;
